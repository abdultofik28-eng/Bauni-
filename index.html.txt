<script>
let current = 0;
const pages = ["lockPage","page1","page2","page3","page4"];

function showPage(i){
  pages.forEach(p=>document.getElementById(p).classList.remove("active"));
  document.getElementById(pages[i]).classList.add("active");
}

function nextPage(){ if(current<pages.length-1){current++;showPage(current);} }
function prevPage(){ if(current>0){current--;showPage(current);} }

function checkPass(){
  let val=document.getElementById("pass").value.toLowerCase();
  if(val==="bauni"){ current=1; showPage(current); }
  else alert("Wrong password 👀");
}

/* IMAGE VIEW */
function openImage(src,title){
  document.getElementById("imageView").style.display="flex";
  document.getElementById("imgSrc").src=src;
  document.getElementById("imgTitle").innerText=title;
  document.querySelector(".nav").classList.add("hidden");
}

function closeImage(){
  document.getElementById("imageView").style.display="none";
  document.querySelector(".nav").classList.remove("hidden");
}

/* SPARKLES */
document.addEventListener("click", e=>{
  let s=document.createElement("div");
  s.className="sparkle";
  s.style.left=e.pageX+"px";
  s.style.top=e.pageY+"px";
  document.body.appendChild(s);
  setTimeout(()=>s.remove(),600);
});
